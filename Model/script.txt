Table User {
  id int [primary key]
  name varchar
  lastname varchar
  user varchar [unique]
  password varchar
  mail varchar [unique]
  rol varchar
}

Table Role {
  id int [primary key]
  RoleType varchar [unique]
}

Table UserRole {
  id int [primary key]
  UserId int [ref: > User.id]
  RoleId int [ref: > Role.id]
}

Table subjects {
  id int [primary key]
  SubjectClasses varchar
}

Table books {
  id int [primary key]
  book varchar
  Content text
  pages int
  author varchar
  User int [ref: > User.id]
  subjectId int [ref: > subjects.id]
}

Table BooksAvailable {
  id int [primary key]
  period varchar
  state varchar
  Books int [ref: > books.id]
}

Table questions {
  id int [primary key]
  questionsBook text
  Answered boolean
  BooksAvailableId int [ref: - BooksAvailable.id]
}

Table Answers {
  id int [primary key]
  AnswerQuestions text
  questionsId int [ref: - questions.id]
  ResultId int [ref: < Result.id]
}

Table responseStudent {
  id int [primary key]
  questionsId int [ref: - questions.id]
  ResultId int [ref: < Result.id]
}

Table Result {
  id int [primary key]
  note decimal
}
